📦 SECTION 1 — DATABASE (DBHelper) ⭐⭐⭐
public class DBHelper extends SQLiteOpenHelper {

    public DBHelper(Context context) {
        super(context, "MyDB", null, 1);
    }

    @Override
    public void onCreate(SQLiteDatabase db) {
        db.execSQL("CREATE TABLE IF NOT EXISTS data(id TEXT, name TEXT, value INTEGER, status TEXT, date TEXT)");
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
        db.execSQL("DROP TABLE IF EXISTS data");
        onCreate(db);
    }

    public void insertData(String id, String name, int value, String status, String date){
        SQLiteDatabase db = this.getWritableDatabase();
        ContentValues cv = new ContentValues();
        cv.put("id", id);
        cv.put("name", name);
        cv.put("value", value);
        cv.put("status", status);
        cv.put("date", date);
        db.insert("data", null, cv);
    }

    public Cursor getAll(){
        SQLiteDatabase db = this.getReadableDatabase();
        return db.rawQuery("SELECT * FROM data", null);
    }

    public Cursor getFiltered(){
        SQLiteDatabase db = this.getReadableDatabase();
        return db.rawQuery("SELECT * FROM data WHERE status='pending'", null);
    }
}
